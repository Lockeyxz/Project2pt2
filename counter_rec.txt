section .text
global _start

_start:
  mov ecx, 0
  call count

  mov eax, 1
  int 0x80

count:
  cmp ecx, 10000
  jge end_count
  inc ecx
  call count
end_count:
  mov eax, 5
  mov ebx, counter_rec_file
  mov ecx, 0x101
  mov edx, 0o666
  int 0x80

  mov eax, 4
  mov ebx, eax
  mov ecx, counter
  mov edx, counter_len
  int 0x80

  mov eax, 6
  mov ebx, eax
  int 0x80

  ret

section .data
  counter_rec_file db "counter_rec.txt", 0
  counter     db "10000", 0
  counter_len equ $ - counter
